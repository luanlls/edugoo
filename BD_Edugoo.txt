use edugoo;

drop table if exists usuarios;
create table usuarios(
	usuario_id				int				auto_increment		primary key,
    usuario_nome			varchar(30),
    usuario_username		varchar(30),
    usuario_senha			varchar(30),
    usuario_tipo			varchar(20)
);
